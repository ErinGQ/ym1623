/*
|--------------------------------------------------------------------------
| EXTEND - 网站的默认主题界面
|--------------------------------------------------------------------------
|
| 每个主题是一个目录，位于网站根目录下，例如主题：default，
| 那么这个主题目录：dirname(FCPATH)."/default/"
|
|	主题目录内保存这个主题所需要资源，例如图片、CSS文件、JS文件等，
|	你的 views 也要对应主题名称，仍以 default 为例，
|	切换 views 目录应该为：APPPATH."views/default/xxx.php"
|
*/
$config['default_theme']	= "default";

/*
|--------------------------------------------------------------------------
| EXTEND - 网站上传文件子目录标记
|--------------------------------------------------------------------------
|
| 在全静态 html 模式中，你需要将用户上传的东西，以这个目录起始
| upload_tag 就是这个小目录的名字
|
*/
$config['upload_tag']	= "upload";

/*
|--------------------------------------------------------------------------
| EXTEND - 网站图片资源的存储子目录标记
|--------------------------------------------------------------------------
|
| 在全静态 html 模式中，你需要将所有图片、CSS、JS等资源，都放在一个目录中，
| image_tag 就是这个小目录的名字
|
|	注意：你的所有主题都应当遵守这个资源标记，比如主题：default，
|	其所有的图片、CSS、JS等资源应当放置在 ./default/img/ 目录中
|
*/
$config['image_tag']	= "img";

/*
|--------------------------------------------------------------------------
| EXTEND - 网站的图片资源连接地址
|--------------------------------------------------------------------------
|
| 当你分离了所有图片文件、JS脚本、CSS文件等，到单独的服务器（俗称负载平衡）
| 只需要设置这个参数即可，无需再修改 views 文件
|
|	例如想迁移 default 主题的资源目录：./default/img/
|	到这个服务器：http://img.负载平衡.com/
|	你需要设置：$config['image_url'] = "http://img.负载平衡.com/";
|
|	注1：如果没使用图片负载平衡，请保持此项为空值
|	注2：想设置的话，路径结尾必须加反斜线 /
|	注3：图片服务器也要分主题目录存储，例如：./default/img/ 的所有资源
|	     将转向到：http://img.负载平衡.com/default/img/ 目录之中
|
*/
$config['image_url']	= "";

/*
|--------------------------------------------------------------------------
| EXTEND - 是否检查 views 和 controller 原文件被修改
|--------------------------------------------------------------------------
|
| 开发网站的时候，每次修改 views 或 controller 文件，其对应的静态 html 文件
| 也需要被刷新，你当然不希望每次都手动删除 html 文件，麻烦还容易删错文件，
| 这个功能完全可以由程序自动探测更新
|
|	设置 html_check_source = true
|	程序将自动检测两个文件的修改状态，免去手动删除已生成的 html 文件的麻烦
|
|	如果网站已开发完成，且不再更换主题文件，请设置 html_check_source = false
|	
|	注意：由于全静态 html 模式是由 js 文件控制 html 更新，你需要刷新两次，
|	      才可以看到新生成的静态 html 文件（就是连续按两次 F5 ~）
|
*/
$config['html_check_source']	= true;
